### 描述

寄生组合继承是一种实现函数或对象之间继承的最佳方式之一，在JavaScript中特别有用。
你需要实现一个 inherit 函数，通过它你可以正确地设置子类型的原型为父类型的一个新实例。

### 代码

```js
//github.com/pearson-g/JavaScript-Guidebook/blob/master/object-oriented-programming/inheritance/parasitic-combination-inheritance.md
// 实现
https: function inherit(Child, Parent) {}

// 测试
function Parent(name) {
  this.name = name;
  this.colors = ['red', 'blue', 'green'];
}

Parent.prototype.sayName = function () {
  console.log(this.name);
};

function Child(name, age) {
  Parent.call(this, name); // 继承属性
  this.age = age;
}

// 使用你的 inherit 函数来配置 Child 的原型
inherit(Child, Parent);

Child.prototype.sayAge = function () {
  console.log(this.age);
};

// 创建一个实例并测试
const child1 = new Child('Kevin', 18);
console.log(child1.name); // Kevin
console.log(child1.colors); // ['red', 'blue', 'green']
child1.sayName(); // Kevin
child1.sayAge(); // 18
```
